Basic program of python of Fibonacci Series

def factorial(n):
    c=1
    for i in range(1,n+1):
        c=c*i
    return c
a=int(input("ENTER NUMBER WHOSE FACTORIAL TO BE FOUND: "))
if a==0:
    print("ANSWER:",a,'! =',1)
elif a>0:
    s=factorial(a)
    print("ANSWER:",a,'! =',s)
else:
    print("ERROR")
