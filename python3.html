Basic program of python for Creating Menu

while True:
#User can add or remove cases
    print("Menu\n 1.Case1\n 2.Case\n 3.Case3\n 4.Exit")
    option=int(input("SELECT FROM 1-4: "))
    if option==1:
        print("CASE1 IS SELECTED")
    elif option==2:
        print("CASE2 IS SELECTED")
    elif option==3:
        print("CASE3 IS SELECTED")
#BREAK statement for
    elif option==4:
        print("EXIT IS SELECTED")
        break
    else:print("!!!ERROR!!!\n SELECT FROM ABOVE OPTIONS!")
